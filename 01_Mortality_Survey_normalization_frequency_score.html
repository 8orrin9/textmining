def gen_DataFrame(df, box_no):
    # Create empty DataFrame
    df_res = pd.DataFrame()

    for i, row in enumerate(df.itertuples()):
        if box_no == 1: word_lst = row.Box1Desc.split()
        else:           word_lst = row.Box2Desc.split()
        df_doc = gen_document(row.ID, word_lst)
        df_res = pd.concat([df_res,df_doc])
        
    df_res.reset_index(inplace=True)
    df_res.drop('index', axis=1, inplace=True)
    
    return df_res

def get_frequency(df_con):

    # Two lists of sentences
    sentences1 = df_con['sentence'].tolist()
    sentences2 = df_kw_m['value'].tolist()
    
    # Initialize array
    ar_freq = np.zeros( (len(sentences1),len(sentences2)) )

    # Count frequency
    for i in range(ar_freq.shape[0]):
        for j in range(ar_freq.shape[1]):
            if sentences1[i] == sentences2[j]:
                ar_freq[i,j] = 1.0

    return ar_freq

def set_score(freq_array, df_con):

    sr_kw = df_kw.count()

    # Initialize
    for key in sr_kw.index:
        df_con[key] = 0.0

    for idx in df_con.index:
        off_b = 0
        off_e = 0
    
        for key in sr_kw.index:
            off_b = off_e
            off_e += sr_kw[key]
            df_con.loc[idx,key] = freq_array[idx][off_b:off_e].sum()
        
        if idx % 1000 == 0:
            print('<cnt:{}, idx:{}> ======================'.format(idx//1000,idx))
        
        # For test...
        #if idx > 1000: break

    print('==== End of job ======================')
    
    return df_con

def get_ratio(_df_Box, _col_st):
    #_df_score = _df_Box[[_col_st] + col_lst_kw].groupby(_col_st).sum().T
    _df_score = _df_Box[[_col_st] + col_lst_kw].groupby(_col_st).mean().T
    _df_ratio = pd.DataFrame(index= _df_score.index)
    for col in _df_score.columns:
        col_r = str(col) + '_r'
        _df_ratio[col_r] = 100 * _df_score[col] / _df_score.sum(axis=1)
    _df_ratio = _df_ratio.round(2)
    
    return _df_score, _df_ratio

def get_ratio2(_df_Box, _col_st_list):
    _df_score = _df_Box[_col_st_list + col_lst_kw].groupby(_col_st_list).mean().T
    _df_ratio = pd.DataFrame(index= _df_score.index)
    for col in _df_score.columns:
        col_r = str(col) + '_r'
        _df_ratio[col_r] = 100 * _df_score[col] / _df_score.sum(axis=1)
    _df_ratio = _df_ratio.round(2)
    
    return _df_score, _df_ratio
